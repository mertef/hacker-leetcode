func maxMin(k: Int, arr: [Int]) -> Int {
    var maxin = Int.max
    var  sorted = arr.sorted()
//    let sdesc = sorted.map {
//        return  "\($0)"
//    }
    print(sdesc)
    for i in 0..<sorted.endIndex {
        if i + k - 1 < sorted.endIndex {
            let max = sorted[i + k - 1]
            let min = sorted[i]
            let dis = max - min
            //print("dis = \(dis)")
            if maxin > dis {
                maxin = dis
            }
        }
    }
    return maxin
    
}
